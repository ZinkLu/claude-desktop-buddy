# E2 CJK 字体支持实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现精简 CJK 字体渲染，使中文 prompt 和 HUD 文本在 X-Knob 设备上正确显示，不再出现 `??`。

**Architecture:** 内置 12×12 bitmap 字体（~350 常用汉字，~7KB ROM），混合渲染引擎（ASCII 走 TFT_eSPI 内置字体，CJK 走 bitmap 查表），UTF-8 aware 文本换行。

**Tech Stack:** ESP32-S3 Arduino, TFT_eSPI, Python 3 + Pillow（仅开发时）

```python
#!/usr/bin/env python3
"""Generate CJK bitmap font header from TTF/OTF source.

Usage:
    python generate_cjk_font.py --chars cjk_chars.txt --font NotoSansCJK.ttf --size 12 --out ../src/font_cjk.h
"""

import argparse
import os
import sys
from pathlib import Path

try:
    from PIL import Image, ImageDraw, ImageFont
except ImportError:
    print("Error: Pillow required. Run: pip install pillow")
    sys.exit(1)


def parse_chars(filepath: str) -> list:
    """Parse char list file, skip comments and empty lines."""
    chars = []
    with open(filepath, 'r', encoding='utf-8') as f:
        for line in f:
            line = line.split('#')[0].strip()
            if not line:
                continue
            for ch in line:
                if ch not in chars and ord(ch) >= 0x4E00:
                    chars.append(ch)
    return chars


def render_glyph(ch: str, font: ImageFont.FreeTypeFont, size: int) -> bytes:
    """Render single char to 1bpp bitmap. Returns bytes array."""
    img = Image.new('1', (size, size), 0)
    draw = ImageDraw.Draw(img)
    
    # Center the glyph
    bbox = draw.textbbox((0, 0), ch, font=font)
    w = bbox[2] - bbox[0]
    h = bbox[3] - bbox[1]
    x = (size - w) // 2 - bbox[0]
    y = (size - h) // 2 - bbox[1]
    
    draw.text((x, y), ch, font=font, fill=1)
    
    # Convert to bytes (row-major, 8 pixels per byte)
    bytes_per_row = (size + 7) // 8
    data = bytearray(bytes_per_row * size)
    for row in range(size):
        for col in range(size):
            if img.getpixel((col, row)):
                byte_idx = row * bytes_per_row + col // 8
                bit_idx = 7 - (col % 8)
                data[byte_idx] |= (1 << bit_idx)
    return bytes(data)


def generate_header(chars: list, font_path: str, size: int, out_path: str):
    """Generate C header file with bitmap font data."""
    if not os.path.exists(font_path):
        print(f"Error: Font file not found: {font_path}")
        sys.exit(1)
    
    font = ImageFont.truetype(font_path, size)
    
    # Sort by codepoint for binary search
    glyphs = []
    for ch in chars:
        cp = ord(ch)
        bitmap = render_glyph(ch, font, size)
        glyphs.append((cp, bitmap))
    glyphs.sort(key=lambda x: x[0])
    
    bytes_per_row = (size + 7) // 8
    
    lines = []
    lines.append("// Auto-generated CJK font data")
    lines.append("// Generated by tools/generate_cjk_font.py")
    lines.append("// DO NOT EDIT MANUALLY")
    lines.append("")
    lines.append("#pragma once")
    lines.append("#include <stdint.h>")
    lines.append("")
    lines.append(f"#define CJK_FONT_SIZE {size}")
    lines.append(f"#define CJK_GLYPH_COUNT {len(glyphs)}")
    lines.append(f"#define CJK_BYTES_PER_GLYPH {bytes_per_row * size}")
    lines.append("")
    
    # Codepoint table
    lines.append("static const uint32_t CJK_CODEPOINTS[] PROGMEM = {")
    for i, (cp, _) in enumerate(glyphs):
        if i % 8 == 0:
            lines.append("    ")
        lines.append(f"0x{cp:04X}, ")
        if i % 8 == 7:
            lines[-1] = lines[-1].rstrip(', ')
            lines.append("")
    if len(glyphs) % 8 != 0:
        lines[-1] = lines[-1].rstrip(', ')
        lines.append("")
    lines.append("};")
    lines.append("")
    
    # Bitmap table
    lines.append("static const uint8_t CJK_BITMAPS[][CJK_BYTES_PER_GLYPH] PROGMEM = {")
    for cp, bitmap in glyphs:
        lines.append(f"    {{ // U+{cp:04X}")
        for row in range(size):
            row_start = row * bytes_per_row
            row_bytes = bitmap[row_start:row_start + bytes_per_row]
            hex_str = ', '.join(f'0x{b:02X}' for b in row_bytes)
            lines.append(f"        {hex_str},")
        lines.append("    },")
    lines.append("};")
    lines.append("")
    
    with open(out_path, 'w', encoding='utf-8') as f:
        f.write('\n'.join(lines))
    
    print(f"Generated {out_path}: {len(glyphs)} glyphs, {bytes_per_row * size} bytes/glyph")
    print(f"Total ROM: {len(glyphs) * bytes_per_row * size} bytes")


def main():
    parser = argparse.ArgumentParser(description='Generate CJK bitmap font header')
    parser.add_argument('--chars', required=True, help='Path to char list file')
    parser.add_argument('--font', required=True, help='Path to TTF/OTF font file')
    parser.add_argument('--size', type=int, default=12, help='Glyph size in pixels (default: 12)')
    parser.add_argument('--out', required=True, help='Output header file path')
    args = parser.parse_args()
    
    chars = parse_chars(args.chars)
    print(f"Loaded {len(chars)} unique CJK chars from {args.chars}")
    
    generate_header(chars, args.font, args.size, args.out)


if __name__ == '__main__':
    main()
```

```cpp
#include "font_cjk.h"
#include <Arduino.h>
#include <TFT_eSPI.h>

// ---------------------------------------------------------------------------
//  Internal state
// ---------------------------------------------------------------------------
static TFT_eSprite* _cjk_target = nullptr;

// ---------------------------------------------------------------------------
//  UTF-8 decoder
// ---------------------------------------------------------------------------

// Decode one UTF-8 codepoint from *pp, advance *pp.
// Returns 0 on invalid sequence.
static uint32_t decode_utf8(const char** pp) {
  const uint8_t* p = (const uint8_t*)*pp;
  uint8_t c = *p;
  if (c == 0) return 0;
  
  if (c < 0x80) {
    *pp = (const char*)(p + 1);
    return c;
  }
  
  if ((c & 0xE0) == 0xC0) {
    // 2-byte sequence
    uint32_t cp = ((c & 0x1F) << 6) | (p[1] & 0x3F);
    *pp = (const char*)(p + 2);
    return cp;
  }
  
  if ((c & 0xF0) == 0xE0) {
    // 3-byte sequence (CJK)
    uint32_t cp = ((c & 0x0F) << 12) | ((p[1] & 0x3F) << 6) | (p[2] & 0x3F);
    *pp = (const char*)(p + 3);
    return cp;
  }
  
  if ((c & 0xF8) == 0xF0) {
    // 4-byte sequence
    uint32_t cp = ((c & 0x07) << 18) | ((p[1] & 0x3F) << 12) | ((p[2] & 0x3F) << 6) | (p[3] & 0x3F);
    *pp = (const char*)(p + 4);
    return cp;
  }
  
  // Invalid lead byte, skip
  *pp = (const char*)(p + 1);
  return 0;
}

// ---------------------------------------------------------------------------
//  Glyph lookup (binary search)
// ---------------------------------------------------------------------------

static int lookup_glyph(uint32_t cp) {
  int lo = 0;
  int hi = CJK_GLYPH_COUNT - 1;
  
  while (lo <= hi) {
    int mid = (lo + hi) >> 1;
    uint32_t mid_cp = pgm_read_dword(&CJK_CODEPOINTS[mid]);
    if (mid_cp == cp) return mid;
    if (mid_cp < cp) lo = mid + 1;
    else hi = mid - 1;
  }
  return -1;
}

// ---------------------------------------------------------------------------
//  Bitmap blit (1bpp → RGB565)
// ---------------------------------------------------------------------------

static void draw_glyph(int x, int y, int glyph_idx, uint16_t color, uint16_t bg) {
  const uint8_t* bitmap = CJK_BITMAPS[glyph_idx];
  const int w = CJK_FONT_SIZE;
  const int h = CJK_FONT_SIZE;
  const int bytes_per_row = (w + 7) >> 3;
  
  for (int row = 0; row < h; row++) {
    for (int col = 0; col < w; col++) {
      int byte_idx = row * bytes_per_row + (col >> 3);
      int bit_idx = 7 - (col & 7);
      uint8_t byte = pgm_read_byte(&bitmap[byte_idx]);
      bool pixel = (byte >> bit_idx) & 1;
      
      if (pixel || bg != TFT_BLACK) {
        uint16_t px_color = pixel ? color : bg;
        _cjk_target->drawPixel(x + col, y + row, px_color);
      }
    }
  }
}

// ---------------------------------------------------------------------------
//  Public API
// ---------------------------------------------------------------------------

void cjk_font_init() {
  // Nothing needed for PROGMEM data
}

void cjk_set_target(TFT_eSprite* sprite) {
  _cjk_target = sprite;
}

void cjk_draw_string(int x, int y, const char* utf8_str,
                     uint16_t color, uint16_t bg, uint8_t font_size) {
  if (!_cjk_target || !utf8_str) return;
  
  int cursor_x = x;
  int cursor_y = y;
  
  // For size 2, scale up the 12x12 glyph to 24x24
  int scale = (font_size == 2) ? 2 : 1;
  int glyph_w = CJK_FONT_SIZE * scale;
  int glyph_h = CJK_FONT_SIZE * scale;
  
  const char* p = utf8_str;
  while (*p) {
    const char* prev_p = p;
    uint32_t cp = decode_utf8(&p);
    if (cp == 0) break;
    
    if (cp < 0x80) {
      // ASCII: use TFT_eSPI built-in font
      _cjk_target->setTextColor(color, bg);
      _cjk_target->setTextSize(font_size);
      _cjk_target->setCursor(cursor_x, cursor_y);
      _cjk_target->print((char)cp);
      
      // Advance cursor (6px per ASCII char at size 1, 12px at size 2)
      cursor_x += 6 * font_size;
    } else {
      // CJK: bitmap lookup
      int glyph_idx = lookup_glyph(cp);
      if (glyph_idx >= 0) {
        if (scale == 1) {
          draw_glyph(cursor_x, cursor_y, glyph_idx, color, bg);
        } else {
          // Scale 2x: draw each pixel as 2x2 block
          const uint8_t* bitmap = CJK_BITMAPS[glyph_idx];
          const int bytes_per_row = (CJK_FONT_SIZE + 7) >> 3;
          for (int row = 0; row < CJK_FONT_SIZE; row++) {
            for (int col = 0; col < CJK_FONT_SIZE; col++) {
              int byte_idx = row * bytes_per_row + (col >> 3);
              int bit_idx = 7 - (col & 7);
              uint8_t byte = pgm_read_byte(&bitmap[byte_idx]);
              bool pixel = (byte >> bit_idx) & 1;
              if (pixel || bg != TFT_BLACK) {
                uint16_t px_color = pixel ? color : bg;
                _cjk_target->fillRect(cursor_x + col * 2, cursor_y + row * 2, 2, 2, px_color);
              }
            }
          }
        }
        cursor_x += glyph_w;
      } else {
        // Missing glyph: draw placeholder box
        _cjk_target->drawRect(cursor_x + 1, cursor_y + 1, glyph_w - 2, glyph_h - 2, color);
        cursor_x += glyph_w;
      }
    }
  }
}

int cjk_text_width(const char* utf8_str, uint8_t font_size) {
  if (!utf8_str) return 0;
  
  int width = 0;
  int scale = (font_size == 2) ? 2 : 1;
  int glyph_w = CJK_FONT_SIZE * scale;
  
  const char* p = utf8_str;
  while (*p) {
    uint32_t cp = decode_utf8(&p);
    if (cp == 0) break;
    
    if (cp < 0x80) {
      width += 6 * font_size;  // ASCII width
    } else {
      width += glyph_w;  // CJK width
    }
  }
  return width;
}

int cjk_utf8_strlen(const char* utf8_str) {
  if (!utf8_str) return 0;
  
  int count = 0;
  const char* p = utf8_str;
  while (*p) {
    uint32_t cp = decode_utf8(&p);
    if (cp != 0) count++;
  }
  return count;
}
```

```cpp
static uint8_t wrapIntoUtf8(const char* in, char out[][48], uint8_t maxRows, uint8_t maxCols) {
  uint8_t row = 0;
  const char* p = in;
  
  while (*p && row < maxRows) {
    uint8_t col = 0;
    uint8_t col_bytes = 0;  // byte position in current row
    
    while (*p && col < maxCols) {
      // Peek at next codepoint
      const char* cp_start = p;
      uint8_t c = (uint8_t)*p;
      int cp_bytes = 1;
      if (c >= 0x80) {
        if ((c & 0xE0) == 0xC0) cp_bytes = 2;
        else if ((c & 0xF0) == 0xE0) cp_bytes = 3;
        else if ((c & 0xF8) == 0xF0) cp_bytes = 4;
      }
      
      // If adding this codepoint exceeds maxCols, break (but always take at least 1)
      if (col + 1 > maxCols && col > 0) break;
      
      // Copy the codepoint bytes
      for (int i = 0; i < cp_bytes && *p; i++) {
        out[row][col_bytes++] = *p++;
      }
      col++;
      
      // If next char is space, consume it but don't count as column
      if (*p == ' ') {
        out[row][col_bytes++] = *p++;
      }
    }
    
    out[row][col_bytes] = 0;
    if (col_bytes > 0) row++;
    else break;  // no progress, avoid infinite loop
  }
  
  return row;
}
```
